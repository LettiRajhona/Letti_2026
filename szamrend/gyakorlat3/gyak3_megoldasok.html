Órai feladat (szűrők fájlban) megoldása:

wc -m minden.txt
wc -w minden.txt
wc -l minden.txt
sort minden.txt
sort minden.txt | uniq -c | sort -nr
sort minden.txt | uniq -c | sort -nr | head -3 | tail -1
grep "alma" minden.txt
grep -w "alma" minden.txt
grep -E '[-0-9+*/^]' minden.txt
cat minden.txt | tr '[:lower:]' '[:upper:]' >> minden_nagybetus.txt
grep '^[a-zA-Z].*[0-9]$' minden.txt
grep -E '^.{10}$' minden.txt
sort -r minden.txt

ls -al | grep '\.' | wc -l
find . -type d

ls -l | cut -d ' ' -f7 | sort -r | head -1
