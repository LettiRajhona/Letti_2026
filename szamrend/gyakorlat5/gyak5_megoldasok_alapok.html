Órai feladat (műveletek) megoldása:

#!/bin/bash

echo "Matematikai műveletek kiíratása, a = 7, b = 4"
a=7
b=4

osszeg=`expr $a + $b`
echo "a + b = $osszeg"

kulonbseg=`expr $a - $b`
echo "a - b = $kulonbseg"

szorzat=`expr $a \* $b`
echo "a * b = $szorzat"

hanyados=`expr $a / $b`
echo "a / b = $hanyados"

maradek=`expr $a % $b`
echo "a % b = $maradek"

hatvany=$((a ** b))
echo "a ^ b = $hatvany"

Órai feladat (P vagy D kezdőbetűk) megoldása:

#!/bin/bash

echo "A könyvtárban található P betűvel kezdődő állományok: "
p=`ls | grep "^[pP].*"`
echo $p

## üres echo: sortörés
echo

echo "A könyvtárban található P vagy D betűvel kezdődő állományok darabszáma: "
pd=`ls | grep -E "^([pP]|[dD]).*" | wc -l`
echo $pd
